SELECT I.SUMMARY, IL.NAME
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.SUMMARY;

SELECT U.FIRSTNAME, U.LASNAME, COUNT(*) AS TOTAL_ISSUES
FROM USERS U JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY U.ID
ORDER BY U.LASNAME, U.FIRSTNAME;
